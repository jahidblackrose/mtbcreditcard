    Database Setup - Oracle Database

    Important: The MTB Credit Card system requires Oracle Database XE 21c or Enterprise Edition. Oracle XE is free and suitable for development.

    System Requirements

    Component Minimum Requirement Recommended
    Operating System Windows 10+, Linux (Ubuntu 20.04+), or macOS 11+ Windows 11, Ubuntu 22.04 LTS

    Disk Space 15 GB free space 30 GB or higher
    Oracle Version Oracle Database XE 21c Oracle Database 21c Enterprise

    Port Requirements

    • 1521 - Oracle Database Listener (default)
    • 8080 - Oracle Application Express (APEX)
    • 5500 - Oracle EM Express (optional management tool)

    Database Schema Setup

    Important: Execute the DDL scripts in the exact order shown below. Scripts are numbered to indicate execution order.

    DDL Script Execution Order

    Order Script File Description Verification Query
    1 00_tables_error_log.sql Error logging table and triggers SELECT * FROM user_tables WHERE table_name = 'TCC_ERROR_LOG';
    2 01_tables_reference.sql Reference data tables (cards, banks, etc.) SELECT COUNT(*) FROM TCC_REF_CARD_PRODUCTS;
    3 02_tables_applicant_application.sql Core application tables SELECT * FROM user_tables WHERE table_name LIKE 'TCC_APP%';
    4 04_tables_session_draft.sql Session and draft management SELECT * FROM user_tables WHERE table_name LIKE 'TCC_SESSION%';
    5 05_tables_limit_issuance_mis.sql Limit and MIS tables SELECT * FROM user_tables WHERE table_name LIKE 'TCC_LIMIT%';
    6 06_tables_extended_entities.sql Extended entity relationships SELECT * FROM user_tables WHERE table_name LIKE 'TCC_EXT%';
    7 pkg_credit_card_application_spec.sql Package specification (60 procedures) SELECT object_name FROM user_procedures WHERE object_name = 'PKG_CREDIT_CARD_APPLICATION';
    8 pkg_credit_card_application_body.sql Package body (implementation) SELECT STATUS FROM user_objects WHERE object_name = 'PKG_CREDIT_CARD_APPLICATION';

    Common Database Issues and Solutions

    Error Message Cause Solution
    ORA-12154: TNS:could not resolve the connect identifier Wrong service name or tnsnames.ora issue Use format: host:port/service_name instead of TNS alias
    ORA-12541: TNS:no listener Oracle listener not running Run: lsnrctl start (Windows) or sudo lsnrctl start (Linux)
    ORA-01017: invalid username/password Wrong credentials or user not created Recreate user using scripts in "Creating Database User" section
    ORA-01950: no privileges on tablespace 'USERS' User lacks tablespace quota Run: GRANT UNLIMITED TABLESPACE TO mtb_credit;
    PLS-00201: identifier must be declared Package not created or invalid Recompile: ALTER PACKAGE PKG_CREDIT_CARD_APPLICATION COMPILE;
    ORA-00942: table or view does not exist DDL scripts not executed in order Re-run scripts starting from missing table number
    DPI-1047: Cannot locate a 64-bit Oracle Client library Oracle Instant Client not installed (Python oracledb) Download Oracle Instant Client from Oracle website or use thick mode: oracledb.init_oracle_client()

    Database Maintenance Commands

    -- Check tablespace usage
SELECT tablespace_name,
       ROUND(bytes/1024/1024, 2) as size_mb,
       ROUND(maxbytes/1024/1024, 2) as max_size_mb,
       ROUND((bytes/maxbytes)*100, 2) as percent_used
FROM user_ts_quotas;

-- Check active sessions
SELECT sid, serial#, status, osuser, machine
FROM v$session
WHERE username = 'MTB_CREDIT';

-- Kill hanging session (use sid, serial# from above)
ALTER SYSTEM KILL SESSION 'sid,serial#';

-- Check long-running queries
SELECT s.sid, s.serial#, s.username, q.sql_text, q.elapsed_time
FROM v$session s, v$sql q
WHERE s.sql_id = q.sql_id
AND s.username = 'MTB_CREDIT'
AND s.status = 'ACTIVE'
ORDER BY q.elapsed_time DESC;

-- Compile invalid objects
BEGIN
    FOR obj IN (SELECT object_name, object_type FROM user_objects WHERE status = 'INVALID')
    LOOP
        EXECUTE IMMEDIATE 'ALTER ' || obj.object_type || ' ' || obj.object_name || ' COMPILE';
    END LOOP;
END;

    Axios Client Setup

    The frontend uses Axios for HTTP requests. Configuration is in src/api/client.ts:

    // File: src/api/client.ts
import axios from 'axios';
import config from '@/config/env';

// Create axios instance
const apiClient = axios.create({
  baseURL: config.API_BASE_URL,
  timeout: 30000,  // 30 seconds
  headers: {
    'Content-Type': 'application/json',
  },
});

// Request interceptor - add auth token
apiClient.interceptors.request.use(
  (config) => {
    const token = localStorage.getItem('auth_token');
    if (token) {
      config.headers.Authorization = `Bearer ${token}`;
    }
    return config;
  },
  (error) => Promise.reject(error)
);

// Response interceptor - handle errors
apiClient.interceptors.response.use(
  (response) => response.data,
  (error) => {
    // Handle common errors
    if (error.response?.status === 401) {
      // Unauthorized - redirect to login
      window.location.href = '/login';
    }
    return Promise.reject(error);
  }
);

export default apiClient;

    Error Log Table Structure

    -- TCC_ERROR_LOG Table
CREATE TABLE TCC_ERROR_LOG (
    error_id          NUMBER GENERATED ALWAYS AS IDENTITY,
    application_id    VARCHAR2(50),
    session_id        VARCHAR2(100),
    error_code        NUMBER(3),          -- 501-560 (procedure codes)
    error_message     CLOB,
    stack_trace       CLOB,
    mobile_number     VARCHAR2(20),
    priority          VARCHAR2(20),       -- HIGH, MEDIUM, LOW
    status            VARCHAR2(50),       -- OPEN, IN_PROGRESS, RESOLVED, CLOSED
    assigned_to       VARCHAR2(50),
    resolution_notes  CLOB,
    created_at        TIMESTAMP DEFAULT SYSTIMESTAMP,
    updated_at        TIMESTAMP DEFAULT SYSTIMESTAMP,
    resolved_at       TIMESTAMP,
    CONSTRAINT pk_error_log PRIMARY KEY (error_id)
);

-- Indexes for support queries
CREATE INDEX idx_error_app ON TCC_ERROR_LOG(application_id);
CREATE INDEX idx_error_mobile ON TCC_ERROR_LOG(mobile_number);
CREATE INDEX idx_error_status ON TCC_ERROR_LOG(status);
CREATE INDEX idx_error_code ON TCC_ERROR_LOG(error_code);
CREATE INDEX idx_error_priority ON TCC_ERROR_LOG(priority);

    Support Query Examples

    1. View All Open Tickets

    SELECT
    error_id,
    application_id,
    mobile_number,
    error_code,
    error_message,
    priority,
    status,
    created_at
FROM TCC_ERROR_LOG
WHERE status IN ('OPEN', 'IN_PROGRESS')
ORDER BY
    priority DESC,
    created_at ASC;

    2. Get Tickets by Mobile Number

    SELECT
    error_id,
    error_code,
    error_message,
    status,
    resolution_notes,
    created_at,
    resolved_at
FROM TCC_ERROR_LOG
WHERE mobile_number = '01712345678'
ORDER BY created_at DESC;

    3. View Error Statistics

    SELECT
    error_code,
    status,
    COUNT(*) as ticket_count,
    ROUND(AVG(
        CASE
            WHEN resolved_at IS NOT NULL
            THEN (resolved_at - created_at) * 24 * 60
            ELSE NULL
        END
    ), 2) as avg_resolution_minutes
FROM TCC_ERROR_LOG
WHERE created_at >= SYSDATE - 30
GROUP BY error_code, status
ORDER BY ticket_count DESC;

    4. Assign Ticket to Staff

    UPDATE TCC_ERROR_LOG
SET
    status = 'IN_PROGRESS',
    assigned_to = 'STAFF_001',
    updated_at = SYSTIMESTAMP
WHERE error_id = 12345
AND status = 'OPEN';

    5. Resolve Ticket

    UPDATE TCC_ERROR_LOG
SET
    status = 'RESOLVED',
    resolution_notes = 'Fixed: Invalid NID format. Corrected and resubmitted.',
    resolved_at = SYSTIMESTAMP,
    updated_at = SYSTIMESTAMP
WHERE error_id = 12345;

    6. Escalate High Priority Issues

    SELECT
    error_id,
    application_id,
    mobile_number,
    error_code,
    error_message,
    created_at
FROM TCC_ERROR_LOG
WHERE priority = 'HIGH'
AND status = 'OPEN'
AND created_at < SYSDATE - (1/24)  -- Older than 1 hour
ORDER BY created_at ASC;
