export async function verifyOTP(mobileNumber: string, otp: string, sessionId: string) {
  if (env.MODE === 'MOCK') {
    await new Promise(resolve => setTimeout(resolve, 500));
    const stored = mockOTPStore[mobileNumber];
    if (!stored) return { status: 400, message: 'OTP not found', data: { verified: false } };
    if (Date.now() > stored.expiresAt) { delete mockOTPStore[mobileNumber]; return { status: 400, message: 'OTP expired', data: { verified: false } }; }
    if (stored.otp !== otp) {
      stored.attempts--;
      if (stored.attempts <= 0) return { status: 429, message: 'Account locked', data: { verified: false, locked: true } };
      return { status: 400, message: `Invalid OTP. ${stored.attempts} attempts remaining.`, data: { verified: false, attempts_remaining: stored.attempts } };
    }
    delete mockOTPStore[mobileNumber];
    return { status: 200, message: 'OTP verified', data: { verified: true, session_id: sessionId, user_exists: true, has_pending_applications: false, applicant_id: 'APP_MOCK_001' } };
  }
  return http.post('/auth/otp/verify', { mobile_number, otp, session_id: sessionId });
}

import { useState, useEffect, useCallback } from 'react';
import { getDraft, saveDraftStep } from '@/api/draft.api';

export function useDraft(sessionId: string | null) {
  const [draft, setDraft] = useState<any>(null);
  const [isLoading, setIsLoading] = useState(false);
  const [lastSaved, setLastSaved] = useState<Date | null>(null);
  const [saveInProgress, setSaveInProgress] = useState(false);

  useEffect(() => {
    const loadDraft = async () => {
      if (!sessionId) { setIsLoading(false); return; }
      setIsLoading(true);
      try {
        const response = await getDraft(sessionId);
        if (response.status === 200 && response.data) {
          setDraft({ currentStep: calculateCurrentStep(response.data.steps || {}), steps: response.data.steps || {} });
        }
      } catch (error) { console.error('Failed to load draft:', error); } finally { setIsLoading(false); }
    };
    loadDraft();
  }, [sessionId]);

  const saveStep = useCallback(async (stepNumber: number, stepName: string, data: any, isComplete: boolean) => {
    if (!sessionId) { console.warn('No session ID'); return; }
    setSaveInProgress(true);
    try {
      const response = await saveDraftStep(sessionId, { session_id: sessionId, step_number: stepNumber, step_name: stepName, data, is_complete: isComplete });
      if (response.status === 200) {
        setLastSaved(new Date());
        setDraft(prev => ({ currentStep: prev?.currentStep || 1, steps: { ...prev?.steps, [stepNumber]: { step_name: stepName, data, is_complete: isComplete, saved_at: new Date().toISOString() } } }));
      }
    } catch (error) { console.error('Failed to save draft:', error); } finally { setSaveInProgress(false); }
  }, [sessionId]);

  return { draft, isLoading, saveInProgress, lastSaved, saveStep, currentStep: draft?.currentStep || 1, getStepData: (stepNumber: number) => draft?.steps[stepNumber]?.data };
}

function calculateCurrentStep(steps: Record<number, any>): number {
  for (let i = 1; i <= 12; i++) { if (!steps[i] || !steps[i].is_complete) { return i; } }
  return 12;
}

  const [currentStep, setCurrentStep] = useState(1);
  const [formData, setFormData] = useState<any>({});

  useEffect(() => {
    if (!isLoading && currentStep > 0) {
      const timer = setTimeout(() => { saveStep(currentStep, `step_${currentStep}`, formData, false); }, 3000);
      return () => clearTimeout(timer);
    }
  }, [formData, currentStep, isLoading]);

  const nextStep = async () => {
    await saveStep(currentStep, `step_${currentStep}`, formData, true);
    setCurrentStep(Math.min(12, currentStep + 1));
  };

  const prevStep = () => { setCurrentStep(Math.max(1, currentStep - 1)); };

  if (isLoading) return <div>Loading application...</div>;

  return (
    <div className="space-y-6">
      <div className="flex justify-between items-center">
        <h2 className="text-2xl font-bold">Step {currentStep} of 12</h2>
        {lastSaved && <span className="text-sm text-muted-foreground">Last saved: {lastSaved.toLocaleTimeString()}</span>}
      </div>
      {currentStep === 1 && <Step1CardSelection data={formData} onChange={setFormData} onNext={nextStep} />}
      {currentStep === 2 && <Step2PersonalInfo data={formData} onChange={setFormData} onNext={nextStep} onPrev={prevStep} />}
      {/* Add remaining steps 3-12 */}
      <div className="flex justify-between">
        <Button onClick={prevStep} disabled={currentStep === 1}>Previous</Button>
        <Button onClick={nextStep}>{currentStep === 12 ? 'Submit' : 'Next'}</Button>
      </div>
    </div>
  );
}
